        class Persona {

            // Contructor destructuring parÃ¡metros variables ES6
            constructor({nombre="SinNombre", apellido="SinApellido", dni="10000000"}) {
                this.nombre = nombre
                this.apellido = apellido
                this.dni = dni
            }

            /**
            *   Metodo get del nombre de la persona 
            *   @return string
            */
            dameNombre() {
                return this.nombre
            }

            dameApellido() {
                return this.apellido
            }

            dameDni() {
                return this.dni
            }

            setNombre(nombre) {
                this.nombre = nombre
            }

            setApellido(apellido) {
                this.apellido = apellido
            }

            setDni(dni) {
                this.dni = dni
            }

            toString() {
                return this.getNombre() + " " + this.getApellido() +" " + this.getDni()
            }

            static personaJson(nombre, apellido, dni) {
                return {
                    nombre,
                    apellido,
                    dni
                }
            }

        }
